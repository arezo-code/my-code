  <script>
        const grid = document.getElementById('loveGrid');
        let revealedCount = 0;

        function initializeGrid() {
            grid.innerHTML = '';
            revealedCount = 0;
            updateCounter();

            for (let i = 0; i < 64; i++) {
                const block = document.createElement('div');
                block.className = 'love-block';
                block.textContent = '♡';
                grid.appendChild(block);

                block.addEventListener('click', function() {
                    if (!this.classList.contains('revealed')) {
                        this.classList.add('revealed');
                        this.textContent = '❤️';
                        revealedCount++;
                        updateCounter();
                    }
                });
            }
        }

        function updateCounter() {
            document.getElementById('counter').textContent = revealedCount;
        }

        function resetGrid() {
            initializeGrid();
        }

        initializeGrid();
    </script>
